Sales Cadence & Consistency

Show code 
library(tidyverse)

sales <- read.csv("data/sales_data.csv")

start_date <- as.Date("2025-06-23")

sales_cadence <- sales %>%
  mutate(
    Date = as.Date(Date, format = "%m/%d/%Y"),
    prev_date = lag(Date, default = start_date),
    days_between = as.numeric(Date - prev_date)
  )

ggplot(sales_cadence, aes(x = days_between)) +
  geom_histogram(
    binwidth = 2,
    boundary = 0,
    fill = "steelblue",
    color = "white"
  ) +
  scale_x_continuous(
    breaks = seq(0, max(sales_cadence$days_between, na.rm = TRUE), by = 2)
  ) +
  scale_y_continuous(
    breaks = seq(0, 14, by = 2),
    expand = expansion(mult = c(0, 0.05))
  ) +
  labs(
    title = "Days Between Consecutive Sales",
    subtitle = "Distribution of gaps between closed deals",
    x = "Days Between Sales",
    y = "Number of Deals"
  ) +
  theme_minimal()

cdjr_makes <- c("Chrysler", "Dodge", "Jeep", "RAM")

cdjr_vs_non <- sales %>%
  mutate(
    dealer_brand = if_else(
      Make %in% cdjr_makes,
      "CDJR",
      "Non-CDJR"
    )
  ) %>%
  count(dealer_brand) %>%
  mutate(
    percent = scales::percent(n / sum(n))

model_counts <- sales %>%
  count(Model, sort = TRUE) %>%
  mutate(
    percent = scales::percent(n / sum(n))
  ) %>%
  rename(
    model = Model,
    deals = n
  )

model_counts_grouped <- model_counts %>%
  mutate(
    model_grouped = if_else(deals == 1, "Other", model)
  ) %>%
  group_by(model_grouped) %>%
  summarise(deals = sum(deals), .groups = "drop") %>%
  arrange(desc(deals))

ggplot(model_counts_grouped,
       aes(x = reorder(model_grouped, deals), y = deals)) +
  geom_col(fill = "steelblue") +
  coord_flip() +
  scale_y_continuous(
    breaks = seq(0, max(model_counts_grouped$deals), by = 1),
    expand = expansion(mult = c(0, 0.05))
  ) +
  labs(
    title = "Vehicle Models Sold",
    subtitle = "Grouped to highlight most frequently sold models",
    x = NULL,
    y = "Number of Deals"
  ) +
  theme_minimal()
